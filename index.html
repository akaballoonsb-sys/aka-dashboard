// =================================================================================== // AKA FLIGHT OPERATIONS & FLEET COMPLIANCE INTEGRATION ENGINE v3.0.0 // =================================================================================== function doGet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var logSheet = ss.getSheetByName("Flight_Logs"); var certSheet = ss.getSheetByName("Certificates"); // Hardcoded real-world baseline hour historical carryovers from physical logs var baselines = { "9M-AKA": 350.33, "9M-BFW": 393.33, "9M-BFV": 392.08, "9M-BGE": 246.08, "9M-BGC": 2.3, "9M-RDY": 0.0 }; var logsData = logSheet.getDataRange().getValues(); var certData = certSheet.getDataRange().getValues(); // 1. PROCESS REAL-TIME FLEET TELEMETRY & INTERVAL RESETS var fleetCalculated = {}; var registrationKeys = ["9M-AKA", "9M-BFW", "9M-BFV", "9M-BGE", "9M-BGC", "9M-RDY"]; var shortNames = {"9M-AKA":"AKA", "9M-BFW":"MOON", "9M-BFV":"PATCH", "9M-BGE":"BOWIE", "9M-BGC":"GAMUDA", "9M-RDY":"RUDY"}; registrationKeys.forEach(function(reg) { fleetCalculated[reg] = { registration: reg, shortName: shortNames[reg], totalHours: baselines[reg], lastServiceHours: baselines[reg], // Default baseline fallback until audit entry found limit: 100 }; }); // Parse logs chronologically to accumulate hours and capture the latest inspection reset mark for (var i = 1; i < logsData.length; i++) { var row = logsData[i]; var companyVal = String(row[2]).trim(); var aircraftReg = String(row[3]).trim().toUpperCase(); // Normalize variant short names to official registration keys if (aircraftReg === "AKA") aircraftReg = "9M-AKA"; if (aircraftReg === "MOON") aircraftReg = "9M-BFW"; if (aircraftReg === "PATCH") aircraftReg = "9M-BFV"; if (aircraftReg === "BOWIE") aircraftReg = "9M-BGE"; if (aircraftReg === "GAMUDA") aircraftReg = "9M-BGC"; if (aircraftReg === "RUDY") aircraftReg = "9M-RDY"; var hUsed = parseFloat(row[4]) || 0; var mUsed = parseFloat(row[5]) || 0; var decimalHours = hUsed + (mUsed / 60); if (fleetCalculated[aircraftReg]) { fleetCalculated[aircraftReg].totalHours += decimalHours; // If row matches your strict dropdown inspection marker, lock this point as the 100hr baseline reset! if (companyVal === "Annual Structural Airworthiness Audit") { fleetCalculated[aircraftReg].lastServiceHours = fleetCalculated[aircraftReg].totalHours; } } } // 2. PARSE REGULATORY COMPLIANCE CERTIFICATES var certificatesArray = []; for (var j = 1; j < certData.length; j++) { var cRow = certData[j]; if (!cRow[0]) continue; var pExpiry = cRow[3] instanceof Date ? cRow[3].toISOString().split('T')[0] : String(cRow[3]); var cExpiry = cRow[4] instanceof Date ? cRow[4].toISOString().split('T')[0] : String(cRow[4]); // Evaluate alerting signatures var today = new Date(); var pDate = new Date(pExpiry); var diffTime = pDate - today; var diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); var permitStatus = "Active"; var regStatus = "Compliant"; if (diffDays <= 0) { permitStatus = "Critical"; regStatus = "Critical"; } else if (diffDays <= 30) { permitStatus = "Warning"; regStatus = "Action Required"; } certificatesArray.push({ registration: String(cRow[0]), serial: String(cRow[2]), permitExpiry: pExpiry, permitStatus: permitStatus, cofrExpiry: cExpiry, status: regStatus }); } // 3. GENERATE ADVANCED OPERATIONAL INSIGHTS (Pilot Analytics) var pilotAnalytics = {}; for (var k = 1; k < logsData.length; k++) { var pRow = logsData[k]; var pilotName = String(pRow[6]).trim(); var h = parseFloat(pRow[4]) || 0; var m = parseFloat(pRow[5]) || 0; var decHrs = h + (m / 60); var bUsed = String(pRow[3]).trim().toUpperCase(); if (!pilotName || pilotName === "" || pRow[2] === "Annual Structural Airworthiness Audit") continue; if (!pilotAnalytics[pilotName]) { pilotAnalytics[pilotName] = { name: pilotName, totalFlights: 0, totalHours: 0, favoriteBalloon: {} }; } pilotAnalytics[pilotName].totalFlights += 1; pilotAnalytics[pilotName].totalHours += decHrs; if (bUsed && bUsed !== "") { pilotAnalytics[pilotName].favoriteBalloon[bUsed] = (pilotAnalytics[pilotName].favoriteBalloon[bUsed] || 0) + 1; } } // Format pilot summary array var insightsArray = Object.keys(pilotAnalytics).map(function(key) { var p = pilotAnalytics[key]; var fav = "None"; var maxCount = 0; Object.keys(p.favoriteBalloon).forEach(function(b) { if (p.favoriteBalloon[b] > maxCount) { maxCount = p.favoriteBalloon[b]; fav = b; } }); return { name: p.name, flights: p.totalFlights, hours: p.totalHours.toFixed(1), favorite: fav }; }); var payload = { fleet: Object.keys(fleetCalculated).map(function(k) { return fleetCalculated[k]; }), certificates: certificatesArray, insights: insightsArray }; return ContentService.createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var logSheet = ss.getSheetByName("Flight_Logs"); var data = JSON.parse(e.postData.contents); var dateObj = new Date(data.date); var months = ["January","February","March","April","May","June","July","August","September","October","November","December"]; var currentMonth = months[dateObj.getMonth()]; // Append standard row schema to Flight_Logs logSheet.appendRow([ currentMonth, data.date, data.company, data.aircraft, parseInt(data.hours) || 0, parseInt(data.minutes) || 0, data.pilot, data.client, data.remarks ]); return ContentService.createTextOutput(JSON.stringify({ status: "Success" })) .setMimeType(ContentService.MimeType.JSON); }
